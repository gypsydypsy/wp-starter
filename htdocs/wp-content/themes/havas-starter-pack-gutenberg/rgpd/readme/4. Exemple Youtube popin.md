// Code modifié

that.linkVideo = that.bloc.querySelectorAll(".articleBox2__mediaLink");
for (var i = 0; i < that.linkVideo.length; i++) {
	that.linkVideo[i].addEventListener("click", function () {
		
		if (tarteaucitron.cookie.get('youtube') == true){						
			that.showVideo($(this).parents(".slider__item").get(0));
		}else{
			if (confirm('YouTube ' + tarteaucitron.lang.fallback + "\n" +  tarteaucitron.lang.disclaimer)){
				tarteaucitron.cookie.set('youtube', true);
				that.showVideo($(this).parents(".slider__item").get(0));
			}
		}		
		
	});
}
