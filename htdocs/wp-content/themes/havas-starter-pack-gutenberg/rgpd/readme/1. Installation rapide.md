* -------------------------------------------------------------------- *
    INSTALLATION POUR LES SITES HDF
* -------------------------------------------------------------------- *

- Créer une branche spéciale rgpd depuis la master (si possible, sinon depuis la dev) : feature/rgpd
- Placer le répertoire tarteaucitron à la racine du thème Wordpress
- Le renommer en /rgpd

# Dans le header
----------------

<script type="text/javascript" src="<?php echo get_template_directory_uri(); ?>/rgpd/tarteaucitron.js"></script>
<script type="text/javascript">
    tarteaucitron.init({
        "hashtag": "#rgpd",         /* Ouvre automatique du panel avec le hashtag */
        "highPrivacy": false,       /* Désactive le consentement implicite en naviguant */
        "allowOnScroll": false,     /* Désactive le consentement au scroll */
        "orientation": "bottom",    /* Positionne le bandeau en haut (top) ou en bas (bottom) ? */
        "adblocker": false,         /* Affiche un message si un adblocker est détecté */
        "showAlertSmall": false,    /* Affiche le petit bandeau en bas à droite */
        "cookieslist": true,        /* Affiche la liste des cookies installés */
        "removeCredit": false       /* Supprime le lien vers la source ? */
    });
</script>


# Dans le footer
----------------

<script type="text/javascript">
	tarteaucitron.user.analyticsUa = 'UA-XXXXXXXX-X';

</script>
